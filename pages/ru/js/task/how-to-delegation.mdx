```html example
<div class="wrapper">
  <div class="child">
    <div>
      <div><div>click me</div></div>
    </div>
  </div>
  <div class="child">
    <div>
      <div><div>click me</div></div>
    </div>
  </div>
  <div class="child">
    <div>
      <div><div>click me</div></div>
    </div>
  </div>
  <div class="other">
    <div>
      <div><div>dont't click me</div></div>
    </div>
  </div>
</div>
```

```js answer
const delegate = (eventName, el, selector, handler) => {
  el.addEventListener(eventName, event => {
    let node = event.target
    const items = [].slice.call(el.querySelectorAll(selector))

    if (items.length) {
      while (node !== el && node !== null) {
        const isTarget = items.some(item => node === item)

        if (isTarget) {
          handler(node)
          break
        } else {
          node = node.parentNode
        }
      }
    }
  })
}

delegate(
  'click',
  document.querySelector('.wrapper'),
  '.child',
  el => (el.style.backgroundColor = 'blue')
)
```
