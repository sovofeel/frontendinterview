```js example
let user = {
  firstName: "Арсений"
};

function func() {
  retrun this.firstName;
}

let funcUser = func.binder(user);
funcUser(); // Арсений
```

```js answer
Function.prototype.binder = function (context, ...argsBind) {
  const fn = this
  return function (...args) {
    return fn.apply(context, argsBind.concat(args))
  }
}
```
