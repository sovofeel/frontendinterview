```jsx example
    // write your code here

  }
  
  function func() {
    return this.firstName;
  }
  
  let funcUser = func.myBind(user)

  return JSON.stringify(funcUser()) // => Арсений

```

```js answer
Function.prototype.myBind = function (context, ...argsBind) {
  const fn = this
  return function (...args) {
    return fn.apply(context, argsBind.concat(args))
  }
}
```
