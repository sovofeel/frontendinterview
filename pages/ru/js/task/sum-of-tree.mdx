```js example
const tree = {
  value: 5,
  right: {
    value: 5,
    right: {
      value: 5,
      right: {
        value: 10,
      },
    },
    left: {
      value: 5,
      left: {
        value: 1,
      },
    },
  },
}

// sumOftree(tree) => 31
```

```js answer
function sumTree(tree) {
  let count = tree.value
  if (tree.left) {
    count += sumTree(tree.left)
  }
  if (tree.right) {
    count += sumTree(tree.right)
  }

  return count
}
```
