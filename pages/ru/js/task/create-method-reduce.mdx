```js example
let arr = [1, 2, 3, 4, 5]

arr.reducer((curr, next) => (curr += next), 0) // => 15
```

```js answer
Array.prototype.reducer = function (func, initState) {
  const arr = this
  const callback = func
  let init = initState

  arr.forEach(function (value, index) {
    init = callback(init, value)
  })
  return init
}
```
