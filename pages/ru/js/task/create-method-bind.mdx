```js answer
if (!Function.prototype.bind) {
  Function.prototype.bind = function (...arg) {
    const func = this
    const context = arg[0]
    const params = arg.slice(1)
    return function (...innerParam) {
      func.apply(context, [...params, ...innerParam])
    }
  }
}
```
